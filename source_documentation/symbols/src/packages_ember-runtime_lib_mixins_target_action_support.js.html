  2 
  3 Ember.TargetActionSupport = Ember.Mixin.create({
  4   target: null,
  5   action: null,
  6 
  7   targetObject: Ember.computed(function() {
  8     var target = get(this, 'target');
  9 
 10     if (Ember.typeOf(target) === "string") {
 11       return Ember.getPath(this, target);
 12     } else {
 13       return target;
 14     }
 15   }).property('target').cacheable(),
 16 
 17   triggerAction: function() {
 18     var action = get(this, 'action'),
 19         target = get(this, 'targetObject');
 20 
 21     if (target && action) {
 22       if (typeof target.send === 'function') {
 23         target.send(action, this);
 24       } else {
 25         if (typeof action === 'string') {
 26           action = target[action];
 27         }
 28         action.call(target, this);
 29       }
 30     }
 31   }
 32 });
